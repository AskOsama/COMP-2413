Complete Schema Script

            
The full SQL script to create the entire University database


            

            

                

                    
-- =============================================
-- University Database Schema
-- =============================================

-- Step 1: Create the database
CREATE DATABASE University;
USE University;

-- Step 2: Create tables without foreign key dependencies first

-- Student table
CREATE TABLE Student (
    Sid     INTEGER PRIMARY KEY,
    sname   VARCHAR(100),
    age     INTEGER,
    sex     VARCHAR(10)
);

-- Professor table
CREATE TABLE Professor (
    Pid     INTEGER PRIMARY KEY,
    pname   VARCHAR(100)
);

-- Step 3: Create tables with foreign key dependencies

-- Course table (depends on Professor)
CREATE TABLE Course (
    Cid     INTEGER PRIMARY KEY,
    cname   VARCHAR(100),
    Pid     INTEGER,
    FOREIGN KEY (Pid) REFERENCES Professor(Pid)
);

-- Marks table (depends on Student and Course)
CREATE TABLE Marks (
    Sid     INTEGER,
    Cid     INTEGER,
    marks   INTEGER,
    PRIMARY KEY (Sid, Cid),
    FOREIGN KEY (Sid) REFERENCES Student(Sid),
    FOREIGN KEY (Cid) REFERENCES Course(Cid)
);

-- =============================================
-- Schema creation complete!
-- =============================================

                

            

        


        
        

            
Key Points

            

                Primary Keys uniquely identify each row

                Foreign Keys create relationships between tables

                Creation Order matters when tables have dependencies
